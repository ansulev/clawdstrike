**2. Create `src/features/right-sidebar/stores/right-sidebar-store.ts`:**

Follow activity-bar-store.ts pattern exactly (create + immer + createSelectors):

```typescript
import { create } from "zustand";
import { immer } from "zustand/middleware/immer";
import { createSelectors } from "@/lib/create-selectors";
import type { RightSidebarPanel } from "../types";

interface RightSidebarState {
  visible: boolean;
  activePanel: RightSidebarPanel;
  width: number;
  actions: {
    toggle: () => void;
    show: () => void;
    hide: () => void;
    setActivePanel: (panel: RightSidebarPanel) => void;
    setWidth: (width: number) => void;
  };
}
```

Defaults: `visible: false`, `activePanel: "speakeasy"`, `width: 320`.

- `toggle`: flip `visible`
- `show`: set `visible = true`
- `hide`: set `visible = false`
- `setActivePanel`: set `activePanel` to given panel
- `setWidth`: clamp between 200 and 480 (`Math.max(200, Math.min(480, width))`)

Export as `useRightSidebarStore` via `createSelectors`.

```typescript
function relativeTime(iso: string): string {
  const diff = Date.now() - new Date(iso).getTime();
  if (diff < 60_000) return "just now";
  if (diff < 3_600_000) return `${Math.floor(diff / 60_000)}m ago`;
  if (diff < 86_400_000) return `${Math.floor(diff / 3_600_000)}h ago`;
  return `${Math.floor(diff / 86_400_000)}d ago`;
}
```
