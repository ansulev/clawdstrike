    ```typescript
    export interface RegistrySearchResult {
      name: string;
      description: string | null;
      latest_version: string | null;  // snake_case matching Rust serde

    export interface RegistrySearchResponse {
      packages: RegistrySearchResult[];
      total: number;
    }

    export interface RegistryVersionSummary {
      version: string;
      pkg_type: string;
      checksum: string;
      yanked: boolean;
      published_at: string;
      downloads: number;
    }

    export interface RegistryPackageInfo {
      name: string;
      description: string | null;
      created_at: string;
      updated_at: string;
      total_downloads: number;
      versions: RegistryVersionSummary[];
    }

    export interface RegistryPopularPackage {
      name: string;
      description: string | null;
      total_downloads: number;
      latest_version: string | null;
    }

    export interface RegistryAttestation {
      publisher_key: string;
      publisher_sig: string;
      registry_key: string | null;
      registry_sig: string | null;
      content_hash: string;
      timestamp: string;
    }
    ```

    **RegistryClient class:**
    ```typescript
    export class RegistryClient {
      private baseUrl: string;

      constructor(baseUrl: string = "http://localhost:8080") {
        this.baseUrl = baseUrl.replace(/\/$/, "");
      }

      async search(query: string, limit = 20, offset = 0): Promise<RegistrySearchResponse>
      async getPackageInfo(name: string): Promise<RegistryPackageInfo>
      async getPopular(limit = 20): Promise<RegistryPopularPackage[]>
      async getPackageStats(name: string): Promise<RegistryPackageStats>
      async getAttestation(name: string, version: string): Promise<RegistryAttestation>
      async getDownloadUrl(name: string, version: string): string
    }
    ```

    Each method does `fetch(this.baseUrl + path)`, checks `response.ok`, parses JSON, and returns the typed result. On network error or non-ok response, return a sensible default (empty results for search, throw for getPackageInfo since that is a specific lookup).

    `getDownloadUrl` is a pure function returning the URL string (not fetching) -- `${this.baseUrl}/api/v1/packages/${encodeURIComponent(name)}/${encodeURIComponent(version)}/download`.

    Export singleton: `export const registryClient = new RegistryClient();`
