    ```typescript
    export interface PluginSandboxProps {
      pluginId: string;
      pluginCode: string;
      css?: string;
      className?: string;
      onReady?: (host: PluginBridgeHost) => void;
      onError?: (error: Error) => void;
    }
    ```

    Implementation:
    1. Use `useRef<HTMLIFrameElement>` for the iframe reference
    2. Use `useRef<PluginBridgeHost>` to hold the bridge host instance
    3. In a `useEffect` (deps: [pluginId, pluginCode]):
       a. Get iframe.contentWindow -- if null, call onError and return
       b. Create `new PluginBridgeHost({ pluginId, targetWindow: iframe.contentWindow })`
       c. Create a message handler: `(event: MessageEvent) => host.handleMessage(event)`
       d. Add the message handler to `window.addEventListener("message", handler)`
       e. Call `onReady?.(host)`
       f. Cleanup function: `window.removeEventListener("message", handler); host.destroy()`
    4. Render: `<div className={className}><iframe ref={iframeRef} sandbox="allow-scripts" srcdoc={buildPluginSrcdoc({ pluginCode, pluginId, css })} style={{ width: "100%", height: "100%", border: "none" }} /></div>`
