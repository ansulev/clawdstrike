**2. use-publication.ts -- extensible publish targets:**

Replace the hardcoded `getAvailableTargets` switch with a registry-based approach:

```typescript
import { getAllPublishTargets } from "./shared-types";
import { getTranslatableTargets } from "./translations";
import { hasAdapter } from "./adapters";

export function getAvailableTargets(fileType: FileType): PublishTarget[] {
  // Built-in targets per file type (preserve existing behavior)
  const builtinTargets: Record<string, PublishTarget[]> = {
    clawdstrike_policy: ["native_policy", "fleet_deploy"],
    sigma_rule: ["native_policy", "fleet_deploy", "json_export", "spl", "kql", "esql"],
    yara_rule: ["json_export"],
    ocsf_event: ["json_export"],
  };

  const targets = builtinTargets[fileType] ?? [];

  // For plugin file types that have an adapter but no built-in targets,
  // offer json_export as a baseline plus any formats reachable via translation.
  if (targets.length === 0 && hasAdapter(fileType)) {
    const translatable = getTranslatableTargets(fileType);
    return ["json_export", ...translatable];
  }

  return targets;
}
```

This keeps existing behavior intact for built-in types while making plugin types automatically gain publish targets based on their translation capabilities.
