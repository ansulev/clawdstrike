```typescript
// ThreatIntelSource interface (FOUND-02) -- exported from plugin SDK or workbench lib
export interface ThreatIntelSource {
  id: string;
  name: string;
  supportedIndicatorTypes: IndicatorType[];
  rateLimit: { requestsPerMinute: number };
  enrich(indicator: Indicator): Promise<EnrichmentResult>;
  healthCheck?(): Promise<boolean>;
}

// Indicator type (FOUND-03)
export type IndicatorType = "hash" | "ip" | "domain" | "url" | "email";

export interface Indicator {
  type: IndicatorType;
  value: string;
  hashAlgorithm?: "md5" | "sha1" | "sha256";
  context?: {
    findingId?: string;
    signalIds?: string[];
  };
}

// EnrichmentResult type (FOUND-04)
export interface ThreatVerdict {
  classification: "malicious" | "suspicious" | "benign" | "unknown";
  confidence: number; // 0-1
  summary: string;
}

export interface EnrichmentResult {
  sourceId: string;
  sourceName: string;
  verdict: ThreatVerdict;
  rawData: Record<string, unknown>;
  mitreTechniques?: string[];
  relatedIndicators?: Indicator[];
  permalink?: string;
  fetchedAt: number;
  cacheTtlMs: number;
}
```

    1. Export a `VIRUSTOTAL_MANIFEST` following the PluginManifest shape from `@clawdstrike/plugin-sdk` types:

    2. Export a `createVirusTotalSource(apiKey: string): ThreatIntelSource` factory:
       - id: "virustotal"
       - name: "VirusTotal"
       - supportedIndicatorTypes: ["hash", "ip", "domain", "url"]
       - rateLimit: { requestsPerMinute: 4 } (VT public API free tier)
       - enrich(indicator) implementation:
         a. Map indicator type to VT v3 endpoint:
            - hash -> `https://www.virustotal.com/api/v3/files/{value}`
            - domain -> `https://www.virustotal.com/api/v3/domains/{value}`
            - ip -> `https://www.virustotal.com/api/v3/ip_addresses/{value}`
            - url -> `https://www.virustotal.com/api/v3/urls/{base64url}` (base64url encode the URL value)
         b. Make fetch() call with header `x-apikey: {apiKey}`
         c. Parse VT v3 response shape: `{ data: { attributes: { last_analysis_stats: { malicious, suspicious, undetected, harmless, ... } } } }`
         d. Normalize to ThreatVerdict:
            - If malicious > 5: classification "malicious", confidence = malicious / total
            - If malicious > 0 and malicious <= 5: classification "suspicious", confidence = 0.3 + (malicious / total) * 0.4
            - If malicious === 0: classification "benign", confidence = harmless / total
            - On any error: classification "unknown", confidence 0
         e. Build summary string: "{malicious}/{total} engines detected as malicious"
         f. Set permalink: "https://www.virustotal.com/gui/file/{value}" (for hash), "...gui/domain/{value}", etc.
         g. Set cacheTtlMs: 300_000 (5 minutes)
       - healthCheck() implementation: GET /api/v3/users/me with the API key, return true if 200

    3. Handle all error paths: HTTP 403 (bad key), 429 (rate limit), network errors, malformed JSON. Never throw -- always return an EnrichmentResult with classification "unknown".

    Write tests first (RED), then implement (GREEN). Tests should mock fetch() using vi.fn() with realistic VT v3 response payloads.
