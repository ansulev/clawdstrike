export class TypedEventEmitter<Events extends Record<string, unknown>> {
  on<K extends keyof Events & string>(event: K, handler: (data: Events[K]) => void): () => void;
  emit<K extends keyof Events & string>(event: K, data: Events[K]): void;
  listenerCount<K extends keyof Events & string>(event: K): number;
  removeAllListeners<K extends keyof Events & string>(event?: K): void;
  dispose(): void;
}
export type SwarmEngineEventMap = {
  "agent.spawned": AgentSpawnedEvent;
  "agent.status_changed": AgentStatusChangedEvent;
  "agent.heartbeat": AgentHeartbeatEvent;
  "agent.terminated": AgentTerminatedEvent;
  ...
};
export interface AgentSpawnedEvent { kind: "agent.spawned"; agent: AgentSession; receipt: Receipt | null; sourceAgentId: string | null; timestamp: number; }
export interface AgentStatusChangedEvent { kind: "agent.status_changed"; agentId: string; previousStatus: AgentSessionStatus; newStatus: AgentSessionStatus; reason: string | null; sourceAgentId: string | null; timestamp: number; }
export interface AgentHeartbeatEvent { kind: "agent.heartbeat"; agentId: string; health: number; workload: number; metricsSnapshot: AgentMetrics; sourceAgentId: string | null; timestamp: number; }
export interface AgentTerminatedEvent { kind: "agent.terminated"; agentId: string; exitCode: number | null; reason: string | null; finalMetrics: AgentMetrics; sourceAgentId: string | null; timestamp: number; }
