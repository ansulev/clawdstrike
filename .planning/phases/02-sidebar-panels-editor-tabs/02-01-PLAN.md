    **Step 1: Add tree helpers to pane-tree.ts**

    Export `replaceNode` (currently module-private) so pane-store.ts can use it. Add two new exported functions:

    ```typescript
    export function addViewToGroup(root: PaneNode, paneId: string, view: PaneView): PaneNode {
      return replaceNode(root, paneId, (node) =>
        node.type === "group"
          ? { ...node, views: [...node.views, view], activeViewId: view.id }
          : node,
      );
    }

    export function removeViewFromGroup(root: PaneNode, paneId: string, viewId: string): PaneNode {
      return replaceNode(root, paneId, (node) => {
        if (node.type !== "group") return node;
        const nextViews = node.views.filter(v => v.id !== viewId);
        let nextActiveId = node.activeViewId;
        if (viewId === node.activeViewId && nextViews.length > 0) {
          const closedIdx = node.views.findIndex(v => v.id === viewId);
          nextActiveId = nextViews[Math.min(closedIdx, nextViews.length - 1)]?.id ?? null;
        }
        return { ...node, views: nextViews, activeViewId: nextActiveId };
      });
    }
    ```
