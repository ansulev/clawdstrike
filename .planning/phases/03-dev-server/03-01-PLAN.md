       ```typescript
       export interface ClawdstrikePluginOptions {
         /** Array of plugin directories to watch. Each entry maps a directory path to a plugin ID. */
         plugins: PluginDevEntry[];
       }

       /** A single plugin development entry: directory + metadata. */
       export interface PluginDevEntry {
         /** Absolute or relative path to the plugin source directory. */
         dir: string;
         /** The plugin's manifest ID (e.g. "clawdstrike.my-guard"). */
         pluginId: string;
         /** The entry point file relative to dir (defaults to "src/index.ts"). */
         entry?: string;
       }

       /** Payload sent over the Vite HMR WebSocket on plugin file change. */
       export interface PluginUpdateEvent {
         /** The ID of the changed plugin. */
         pluginId: string;
         /** Absolute path to the plugin's entry file. */
         entryPath: string;
         /** Unix timestamp (ms) when the change was detected. */
         timestamp: number;
       }

       /** The custom HMR event name used for plugin updates. */
       export const PLUGIN_UPDATE_EVENT = 'clawdstrike:plugin-update' as const;
       ```

    1. Create `src/watcher.ts` with `setupPluginWatcher(server: ViteDevServer, fileMap: FilePluginMap, options: ClawdstrikePluginOptions)`:
       - For each `PluginDevEntry` in `options.plugins`, resolve the dir to absolute path (using `path.resolve`) and register in `fileMap`
       - Use `server.watcher.add(dir)` to add each plugin directory to chokidar's watch list (Vite already has a chokidar watcher)
       - Listen to `server.watcher` for `"change"` events. On change:
         a. Call `fileMap.resolve(changedFilePath)` to get the pluginId
         b. If pluginId found, call `server.ws.send(PLUGIN_UPDATE_EVENT, { pluginId, entryPath: fileMap.getEntry(pluginId), timestamp: Date.now() } satisfies PluginUpdateEvent)`
         c. If pluginId not found, ignore the change (it's not a plugin file)
       - Also listen for `"add"` events (new files) with same logic
       - Log watched directories to console: `[clawdstrike] Watching plugin: ${pluginId} at ${dir}`

    2. Create `src/index.ts` as the main export:
       ```typescript
       import type { Plugin } from 'vite';
       import type { ClawdstrikePluginOptions } from './types';
       import { FilePluginMap } from './file-plugin-map';
       import { setupPluginWatcher } from './watcher';

       export type { ClawdstrikePluginOptions, PluginDevEntry, PluginUpdateEvent } from './types';
       export { PLUGIN_UPDATE_EVENT } from './types';
       export { FilePluginMap } from './file-plugin-map';

       export function clawdstrikePlugin(options: ClawdstrikePluginOptions): Plugin {
         const fileMap = new FilePluginMap();
         return {
           name: 'vite-plugin-clawdstrike',
           configureServer(server) {
             setupPluginWatcher(server, fileMap, options);
           },
         };
       }

       export default clawdstrikePlugin;
       ```
