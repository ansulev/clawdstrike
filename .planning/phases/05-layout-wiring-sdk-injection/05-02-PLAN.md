2. Add a private helper method `buildViewsApi(pluginId: string, disposables: Disposable[])` to the PluginLoader class that constructs the views object:

```typescript
private buildViewsApi(pluginId: string, disposables: Disposable[]): PluginActivationContext["views"] {
  const resolveComponent = (comp: ComponentType<any> | (() => Promise<{ default: ComponentType<any> }>)): ComponentType<any> => {
    if (typeof comp === "function" && comp.length === 0) {
      // Heuristic: zero-arg function that is NOT a React component constructor is a lazy factory
      // Wrap in React.lazy for deferred loading
      try {
        return lazy(comp as () => Promise<{ default: ComponentType<any> }>);
      } catch {
        return comp as ComponentType<any>;
      }
    }
    return comp as ComponentType<any>;
  };

  return {
    registerEditorTab(contribution) {
      const viewId = `${pluginId}.${contribution.id}`;
      const dispose = registerView({
        id: viewId,
        slot: "editorTab",
        label: contribution.label,
        icon: contribution.icon,
        component: resolveComponent(contribution.component),
      });
      disposables.push(dispose);
      return dispose;
    },
    registerBottomPanelTab(contribution) {
      const viewId = `${pluginId}.${contribution.id}`;
      const dispose = registerView({
        id: viewId,
        slot: "bottomPanelTab",
        label: contribution.label,
        icon: contribution.icon,
        component: resolveComponent(contribution.component),
      });
      disposables.push(dispose);
      return dispose;
    },
    registerRightSidebarPanel(contribution) {
      const viewId = `${pluginId}.${contribution.id}`;
      const dispose = registerView({
        id: viewId,
        slot: "rightSidebarPanel",
        label: contribution.label,
        icon: contribution.icon,
        component: resolveComponent(contribution.component),
      });
      disposables.push(dispose);
      return dispose;
    },
    registerStatusBarWidget(contribution) {
      const viewId = `${pluginId}.${contribution.id}`;
      const dispose = statusBarRegistry.register({
        id: viewId,
        side: contribution.side,
        priority: contribution.priority,
        render: () => {
          const Comp = resolveComponent(contribution.component);
          return createElement(Comp, { viewId });
        },
      });
      disposables.push(dispose);
      return dispose;
    },
  };
}
```

The bug: ActivityBarItemContribution has `href` which is a route path (e.g. "/guards"), but the PluginLoader passes it to `this.resolveViewEntrypoint(item.href)` which tries to `import(href)` -- this fails for route paths.

Fix in 3 files:

**1. apps/workbench/src/lib/plugins/types.ts** -- Add `entrypoint` field to ActivityBarItemContribution:

```typescript
export interface ActivityBarItemContribution {
  id: string;
  section: string;
  label: string;
  icon: string;
  /** Route path for navigation (used for built-in nav items). */
  href: string;
  /** Module path for lazy-loaded panel component. Used instead of href for view loading. */
  entrypoint?: string;
  order?: number;
}
```

**2. packages/sdk/plugin-sdk/src/types.ts** -- Same change to SDK copy:

```typescript
export interface ActivityBarItemContribution {
  id: string;
  section: string;
  label: string;
  icon: string;
  /** Route path for navigation. */
  href: string;
  /** Module path for lazy-loaded panel component. Used instead of href for view loading. */
  entrypoint?: string;
  order?: number;
}
```

**3. apps/workbench/src/lib/plugins/plugin-loader.ts** -- Fix the activity bar routing (around line 607-619):

Change from:
```typescript
component: lazy(() => this.resolveViewEntrypoint(item.href)),
```

To:
```typescript
component: item.entrypoint
  ? lazy(() => this.resolveViewEntrypoint(item.entrypoint!))
  : (() => null) as ComponentType<any>,  // No entrypoint = route-based nav, no lazy component
```

Also update the meta to include entrypoint:
```typescript
meta: { section: item.section, href: item.href, entrypoint: item.entrypoint },
```

This way:
- Plugins with `entrypoint` get a lazy-loaded component (correct behavior)
- Plugins with only `href` (route-based navigation) get a null component placeholder, and the DesktopLayout routes handle rendering (existing behavior from Phase 4)
