       const engine = engineRef.current;

       if (!engine) {

       const guardAction: GuardedAction = {

         context: {
           operation: "agent_spawn",
           launchClaude: opts.launchClaude ?? false,
           command: opts.command,
         },
         requestedAt: Date.now(),
       };

       // Step 2: Evaluate guard pipeline (fail-closed if no evaluator)
       const result = await engine.evaluateGuard(guardAction);

       // Step 3: If denied, create receipt node only (no session spawned)
       if (!result.allowed) {
         // Find a position for the deny receipt
         const position = opts.position ?? {
           x: 100 + Math.random() * 400,
           y: 100 + Math.random() * 300,

           nodeType: "receipt",

               guard: gr.guardId,
               allowed: gr.allowed,
               duration_ms: gr.durationMs,
             })),
             signature: result.receipt.signature,
             publicKey: result.receipt.publicKey,
             status: "completed",
             engineManaged: true,
           },
         });
         return denyReceiptNode;
       }

       // We do NOT call spawnSession from session context here because
