2. **Add file loading effect**: When FileEditorShell mounts with a filePath that has no matching tab in policy-tabs-store, it needs to load the file content and create the tab:
   ```typescript
   useEffect(() => {
     if (!filePath || tabMeta) return; // Already loaded or no path
     if (filePath.startsWith("__new__/")) return; // Untitled file, skip loading

     // Load file content via Tauri bridge
     async function loadFile() {
       try {
         const { readDetectionFileByPath } = await import("@/lib/workbench/tauri-bridge");
         const result = await readDetectionFileByPath(filePath!);
         if (result) {
           usePolicyTabsStore.getState().openTabOrSwitch(
             result.path,
             result.fileType,
             result.content,
             filePath!.split("/").pop() ?? "File",
           );
         }
       } catch (err) {
         console.warn("[FileEditorShell] Failed to load file:", filePath, err);
       }
     }
     loadFile();
   }, [filePath, tabMeta]);
   ```

3. **Handle __new__ routes**: For untitled files (route `/file/__new__/<tabId>`), look up the tab by the tabId in the route:

   // Determine tabMeta: either by filePath match or by __new__ tabId
   const isNewFile = filePath?.startsWith("__new__/") ?? false;
   const newTabId = isNewFile ? filePath!.split("/")[1] : null;

   const tabMeta = usePolicyTabsStore(
     (s) => {
       if (isNewFile && newTabId) {
         return s.tabs.find((t) => t.id === newTabId) ?? null;

       return s.tabs.find((t) => t.filePath === filePath) ?? null;
