    **Path normalization:**
    ```rust
    pub fn normalize_path(path: &str) -> String {
        let path = path.replace('\\', "/");
        // Strip leading drive letter (C:/)
        let path = if path.len() >= 3 && path.as_bytes()[1] == b':' && path.as_bytes()[2] == b'/' {
            &path[3..]
        } else {
            &path
        };
        // Strip leading slash
        let path = path.strip_prefix('/').unwrap_or(path);
        // Find workspace-relative portion: look for common workspace markers
        // Strategy: strip everything up to and including the last occurrence of
        // known project root indicators, or return as-is if none found.
        // For simplicity, strip any prefix ending with a path component that
        // looks like a project root (contains .git, or user home paths).
        // The practical approach: the CLIENT should send workspace-relative paths.
        // The server just strips leading slashes and drive letters as a safety net.
        path.to_string()
    }
    ```
    The normalization is intentionally simple — the primary normalization responsibility is on the client (Phase 19 will strip projectRoot). The server's job is to be a safety net: strip absolute path prefixes so that different client-side bugs don't create duplicate rooms.

    pub async fn spawn_heartbeat_reaper(hub: Arc<PresenceHub>, shutdown: Arc<Notify>) {
        let mut interval = tokio::time::interval(Duration::from_secs(REAPER_INTERVAL_SECS));
        loop {
            tokio::select! {
                _ = interval.tick() => {
                    let stale = hub.stale_analysts(Duration::from_secs(HEARTBEAT_TTL_SECS));
                    for fingerprint in stale {
                        if let Some(info) = hub.leave(&fingerprint) {
                            tracing::info!(fingerprint = %fingerprint, "Evicting stale analyst");
                            hub.broadcast(ServerMessage::AnalystLeft { fingerprint });
                        }
                    }
                }
                _ = shutdown.notified() => {
                    tracing::info!("Heartbeat reaper shutting down");
                    break;
                }
            }
        }
    }

    **WebSocket handler:**
    ```rust
    #[derive(Deserialize)]
    pub struct PresenceQuery {
        pub token: Option<String>,
    }

    pub async fn ws_handler(
        ws: WebSocketUpgrade,
        State(state): State<AppState>,
        Query(query): Query<PresenceQuery>,
    ) -> Result<impl IntoResponse, V1Error> {
        // Auth: validate token from query param (WS can't use custom headers)
        if state.auth_enabled() {
            let token = query.token.as_deref().ok_or_else(|| {
                V1Error::unauthorized("MISSING_TOKEN", "WebSocket requires ?token= query parameter")
            })?;
            state.auth_store.validate_key(token).await.map_err(|_| {
                V1Error::unauthorized("INVALID_TOKEN", "Invalid or expired token")
            })?;
        }

        Ok(ws.on_upgrade(move |socket| handle_ws(socket, state)))
    }

    async fn handle_ws(socket: WebSocket, state: AppState) {
        let (mut sender, mut receiver) = socket.split();
        let hub = &state.presence_hub;
        let mut rx = hub.subscribe();
        let mut analyst_fingerprint: Option<String> = None;

        // Spawn task to forward broadcast messages to this client's WS sender
        let (internal_tx, mut internal_rx) = tokio::sync::mpsc::channel::<ServerMessage>(64);
        let send_task = tokio::spawn(async move {
            while let Some(msg) = internal_rx.recv().await {
                if let Ok(text) = serde_json::to_string(&msg) {
                    if sender.send(Message::Text(text.into())).await.is_err() {
                        break;
                    }
                }
            }
        });

        // Forward broadcast to internal channel (filtered to this client's rooms)
        let hub_clone = state.presence_hub.clone();
        let internal_tx_clone = internal_tx.clone();
        let forward_task = tokio::spawn(async move {
            while let Ok(msg) = rx.recv().await {
                // Forward all messages — client-side filtering is more flexible
                // but we could filter by room here for optimization
                let _ = internal_tx_clone.send(msg).await;
            }
        });

        // Process incoming messages
        while let Some(Ok(msg)) = receiver.next().await {
            match msg {
                Message::Text(text) => {
                    match serde_json::from_str::<ClientMessage>(&text) {
                        Ok(client_msg) => {
                            match client_msg {
                                ClientMessage::Join { fingerprint, display_name, sigil } => {
                                    let info = hub.join(&fingerprint, &display_name, &sigil);
                                    analyst_fingerprint = Some(fingerprint.clone());
                                    // Send welcome with current roster
                                    let _ = internal_tx.send(ServerMessage::Welcome {
                                        analyst_id: fingerprint.clone(),
                                        color: info.color.clone(),
                                        roster: hub.roster(),
                                    }).await;
                                    // Broadcast join to others
                                    hub.broadcast(ServerMessage::AnalystJoined { analyst: info });
                                }
                                ClientMessage::ViewFile { file_path } => {
                                    if let Some(ref fp) = analyst_fingerprint {
                                        let normalized = normalize_path(&file_path);
                                        hub.view_file(fp, &file_path);
                                        hub.broadcast(ServerMessage::AnalystViewing {
                                            fingerprint: fp.clone(),
                                            file_path: normalized,
                                        });
                                    }
                                }
                                ClientMessage::LeaveFile { file_path } => {
                                    if let Some(ref fp) = analyst_fingerprint {
                                        let normalized = normalize_path(&file_path);
                                        hub.leave_file(fp, &file_path);
                                        hub.broadcast(ServerMessage::AnalystLeftFile {
                                            fingerprint: fp.clone(),
                                            file_path: normalized,
                                        });
                                    }
                                }
                                ClientMessage::Cursor { file_path, line, ch } => {
                                    if let Some(ref fp) = analyst_fingerprint {
                                        hub.touch(fp);
                                        hub.broadcast(ServerMessage::AnalystCursor {
                                            fingerprint: fp.clone(),
                                            file_path: normalize_path(&file_path),
                                            line, ch,
                                        });
                                    }
                                }
                                ClientMessage::Selection { file_path, anchor_line, anchor_ch, head_line, head_ch } => {
                                    if let Some(ref fp) = analyst_fingerprint {
                                        hub.touch(fp);
                                        hub.broadcast(ServerMessage::AnalystSelection {
                                            fingerprint: fp.clone(),
                                            file_path: normalize_path(&file_path),
                                            anchor_line, anchor_ch, head_line, head_ch,
                                        });
                                    }
                                }
                                ClientMessage::Heartbeat => {
                                    if let Some(ref fp) = analyst_fingerprint {
                                        hub.touch(fp);
                                        let _ = internal_tx.send(ServerMessage::HeartbeatAck).await;
                                    }
                                }
                            }
                        }
                        Err(e) => {
                            let _ = internal_tx.send(ServerMessage::Error {
                                message: format!("Invalid message: {e}"),
                            }).await;
                        }
                    }
                }
                Message::Close(_) => break,
                _ => {} // ignore binary/ping/pong
            }
        }

        // Cleanup on disconnect
        if let Some(fp) = analyst_fingerprint {
            hub.leave(&fp);
            hub.broadcast(ServerMessage::AnalystLeft { fingerprint: fp });
        }
        forward_task.abort();
        send_task.abort();
    }
    ```
