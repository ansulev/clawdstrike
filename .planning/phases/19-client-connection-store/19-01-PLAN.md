```typescript
  private reconnectTimer: ReturnType<typeof setTimeout> | null = null;

  private isReconnect = false;

  constructor(opts: FleetEventStreamOptions) { this.opts = opts; }
  connect(): void { /* ... */ }
  disconnect(): void { /* ... */ }
  private scheduleReconnect(): void {
    const delay = BACKOFF_DELAYS[Math.min(this.reconnectAttempt, BACKOFF_DELAYS.length - 1)];
    this.reconnectAttempt++;
    this.isReconnect = true;
    this.reconnectTimer = setTimeout(() => { this.reconnectTimer = null; this.connect(); }, delay);
```

```typescript
import { resolveProxyBase } from "@/components/workbench/editor/live-agent-tab";
import type { ClientMessage, PresenceConnectionState, PresenceSocketOptions, ServerMessageRaw } from "./types";
import { HEARTBEAT_INTERVAL_MS } from "./types";

const BACKOFF_BASE_DELAYS = [1000, 2000, 4000, 8000, 16000];

export class PresenceSocket {
  private opts: PresenceSocketOptions;
  private ws: WebSocket | null = null;
  private reconnectTimer: ReturnType<typeof setTimeout> | null = null;
  private heartbeatTimer: ReturnType<typeof setInterval> | null = null;
  private reconnectAttempt = 0;
  private isReconnect = false;
  private disposed = false;

  constructor(opts: PresenceSocketOptions) { this.opts = opts; }

  connect(): void { ... }
  disconnect(): void { ... }
  send(msg: ClientMessage): void { ... }

  private onOpen(): void { ... }
  private onMessage(event: MessageEvent): void { ... }
  private onClose(): void { ... }
  private onError(): void { ... }
  private scheduleReconnect(): void { ... }
  private startHeartbeat(): void { ... }
  private stopHeartbeat(): void { ... }
  private sendJoin(): void { ... }
}
```

**connect() method:**
1. Set state to "connecting" via `opts.onStateChange("connecting")`
2. Get API key: `const apiKey = this.opts.getApiKey()`
3. If no apiKey, set state to "disconnected" and return (graceful offline degradation)
4. Resolve URL: `const base = resolveProxyBase(this.opts.hushdUrl)`
5. Build WS URL: `const wsUrl = base.replace(/^http/, "ws") + "/api/v1/presence?token=" + encodeURIComponent(apiKey)`
6. Create `new WebSocket(wsUrl)` and attach onopen/onmessage/onclose/onerror handlers
7. On errors during construction, catch and schedule reconnect

**onOpen() handler:**
1. Reset reconnectAttempt to 0
2. Set state to "connected"
3. Call `sendJoin()` to send the Join message with operator identity
4. Start heartbeat timer
5. If `this.isReconnect`, call `opts.onReconnect()` then set `isReconnect = false`

**sendJoin() method:**
1. Get identity from `this.opts.getIdentity()`
2. If no identity, log warning and return (can't join without identity)
3. Send Join message: `this.send({ type: "join", fingerprint: identity.fingerprint, display_name: identity.displayName, sigil: identity.sigil })`

**onMessage(event) handler:**
1. Parse `JSON.parse(event.data)` as `ServerMessageRaw`
2. Call `this.opts.onMessage(parsed)`
3. Wrap in try/catch — skip malformed messages (log warning, do not tear down connection)

**onClose() handler:**
1. Stop heartbeat timer
2. If not disposed, set state to "reconnecting" and call `scheduleReconnect()`

**onError() handler:**
1. Log warning, but do NOT set state here — onClose fires after onError and handles state transition

**scheduleReconnect() method** (DIFFERS from FleetEventStream — adds jitter):
1. `const baseDelay = BACKOFF_BASE_DELAYS[Math.min(this.reconnectAttempt, BACKOFF_BASE_DELAYS.length - 1)]`
2. `const jitter = baseDelay * (0.5 + Math.random() * 0.5)` — 50-100% of base delay
3. `this.reconnectAttempt++`
4. `this.isReconnect = true`
5. Set reconnectTimer to call `this.connect()` after jitter delay

**startHeartbeat() method:**
1. Stop any existing heartbeat timer
2. `this.heartbeatTimer = setInterval(() => this.send({ type: "heartbeat" }), HEARTBEAT_INTERVAL_MS)`

**stopHeartbeat() method:**
1. If heartbeatTimer, clear it and set to null

**send(msg) method:**
1. If `this.ws?.readyState === WebSocket.OPEN`, call `this.ws.send(JSON.stringify(msg))`
2. Otherwise silently drop (no error — offline degradation)

**disconnect() method:**
1. Set `this.disposed = true`
2. Stop heartbeat
3. Clear reconnect timer
4. Close WebSocket with code 1000 (normal closure)
5. Set state to "disconnected"
6. Reset reconnectAttempt

**CRITICAL: Do NOT import tauri-plugin-websocket. Use native browser `WebSocket` API only.**
**CRITICAL: getApiKey() is called fresh on every connect()/reconnect, NOT cached at construction time.**
