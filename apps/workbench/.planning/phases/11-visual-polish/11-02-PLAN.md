    ```typescript
    const projectRoots = useProjectStore((s) => s.projectRoots);
    let relPath = filePath;
    for (const root of projectRoots) {
      if (filePath.startsWith(root + "/")) {
        relPath = filePath.slice(root.length + 1);
        break;
      }
      if (filePath.startsWith(root)) {
        relPath = filePath.slice(root.length).replace(/^\//, "");
        break;
      }

    if (relPath === filePath && project?.rootPath) {
      if (filePath.startsWith(project.rootPath + "/")) {
        relPath = filePath.slice(project.rootPath.length + 1);
      } else if (filePath.startsWith(project.rootPath)) {
        relPath = filePath.slice(project.rootPath.length).replace(/^\//, "");
      }
    }

    const segments = relPath.split("/").filter(Boolean);
    ```

The safer approach is to re-scan the root from disk after each mutation, ensuring the tree always reflects reality.

    **Fix approach:** After each mutation in `createFile`, `renameFile`, and `deleteFile`, call `loadRoot(project.rootPath)` to re-scan the directory from disk. This is a fire-and-forget async call. The in-memory tree update remains for immediate UI feedback; the loadRoot re-scan corrects any discrepancies within milliseconds.
